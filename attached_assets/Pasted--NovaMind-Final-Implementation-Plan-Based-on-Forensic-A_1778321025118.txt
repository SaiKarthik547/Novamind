"""

14-table SQLite episodic + semantic memory system.
WAL journal mode — crash-safe writes, no full fsync overhead.

import time
from datetime import datetime
from pathlib import Path
from typing import Any, Dict, List, Optional

logger = logging.getLogger("MemorySystem")

DB_PATH = os.path.expanduser("~/.novamind/memory.db")

class MemorySystem:
    def __init__(self, db_path: str = DB_PATH):
        self._db_path = db_path
        self._lock = threading.Lock()
        self._session_id = f"session_{int(time.time())}"
        os.makedirs(os.path.dirname(db_path), exist_ok=True)
        self._init_db()
        self._start_session()
        logger.info(f"MemorySystem ready: {db_path}")

    def _conn(self) -> sqlite3.Connection:
        conn = sqlite3.connect(self._db_path, timeout=10)
        conn.row_factory = sqlite3.Row
        return conn

    def _init_db(self) -> None:
        with self._lock:
            with self._conn() as conn:
                conn.executescript(SCHEMA_SQL)

    def _start_session(self) -> None:
        with self._lock:
            with self._conn() as conn:
                conn.execute(
                    "INSERT OR IGNORE INTO sessions (session_id, started_at) VALUES (?,?)",
                    (self._session_id, datetime.now().isoformat())
                )

    # ── Task lifecycle ────────────────────────────────────────────────────────

    def record_task_start(self, task_id: str, request: str,
                          task_type: str = "", risk_level: str = "",
                          summary: str = "", total_steps: int = 0) -> None:
        with self._lock:
            with self._conn() as conn:
                conn.execute("""
                    INSERT OR REPLACE INTO tasks
                    (task_id, session_id, request, task_type, risk_level,
                     status, summary, total_steps, started_at)
                    VALUES (?,?,?,?,?,?,?,?,?)
                """, (task_id, self._session_id, request, task_type,
                      risk_level, "running", summary, total_steps,
                      datetime.now().isoformat()))

    def record_task_end(self, task_id: str, status: str,
                        steps_ok: int = 0, steps_fail: int = 0,
                        error_summary: str = None) -> None:
        with self._lock:
            with self._conn() as conn:
                conn.execute("""
                    UPDATE tasks SET status=?, steps_ok=?, steps_fail=?,
                    error_summary=?, ended_at=? WHERE task_id=?
                """, (status, steps_ok, steps_fail, error_summary,
                      datetime.now().isoformat(), task_id))

    def record_step_start(self, task_id: str, step_number: int,
                          description: str, agent: str, action: str,
                          parameters: Dict) -> Optional[int]:
        with self._lock:
            with self._conn() as conn:
                cur = conn.execute("""
                    INSERT INTO steps
                    (task_id, step_number, description, agent, action,
                     parameters, status, started_at)
                    VALUES (?,?,?,?,?,?,?,?)
                """, (task_id, step_number, description, agent, action,
                      json.dumps(parameters, default=str),
                      "running", datetime.now().isoformat()))
                return cur.lastrowid

    def record_step_end(self, step_db_id: int, status: str,
                        output: str = "", error: str = "",
                        retry_count: int = 0) -> None:
        with self._lock:
            with self._conn() as conn:
                conn.execute("""
                    UPDATE steps SET status=?, output=?, error=?,
                    retry_count=?, ended_at=? WHERE id=?
                """, (status, output[:3000], error[:1000],
                      retry_count, datetime.now().isoformat(), step_db_id))

    def record_agent_action(self, task_id: str, agent: str, action: str,
                            parameters: Dict, result: Dict,
                            step_id: int = None) -> None:
        with self._lock:
            with self._conn() as conn:
                conn.execute("""
                    INSERT INTO tool_results
                    (task_id, step_id, agent, action, parameters,
                     result, success, created_at)
                    VALUES (?,?,?,?,?,?,?,?)
                """, (task_id, step_id, agent, action,
                      json.dumps(parameters, default=str)[:2000],
                      json.dumps(result, default=str)[:2000],
                      int(result.get("success", False)),
                      datetime.now().isoformat()))

    def log_error(self, error_msg: str, task_id: str = "",
                  agent: str = "", action: str = "") -> None:
        with self._lock:
            with self._conn() as conn:
                conn.execute("""
                    INSERT INTO errors (task_id, agent, action, error_msg, created_at)
                    VALUES (?,?,?,?,?)
                """, (task_id, agent, action, error_msg[:2000],
                      datetime.now().isoformat()))

    def store_experience(self, experience: Dict) -> None:
        with self._lock:
            with self._conn() as conn:
                conn.execute("""
                    INSERT INTO episodes
                    (task, task_type, steps_ok, steps_fail, success,
                     error, duration, timestamp)
                    VALUES (?,?,?,?,?,?,?,?)
                """, (
                    experience.get("task", "")[:500],
                    experience.get("task_type", ""),
                    experience.get("steps_ok", 0),
                    experience.get("steps_fail", 0),
                    int(experience.get("success", False)),
                    experience.get("error", ""),
                    experience.get("duration", 0.0),
                    experience.get("timestamp", datetime.now().isoformat()),
                ))

    def find_similar_experiences(self, query: str,
                                 limit: int = 3) -> List[Dict]:
        """Substring search fallback. Semantic search requires sentence_transformers."""
        with self._lock:
            with self._conn() as conn:
                rows = conn.execute("""
                    SELECT task, task_type, steps_ok, steps_fail,
                           success, error, duration, timestamp
                    FROM episodes
                    WHERE task LIKE ?
                    ORDER BY timestamp DESC LIMIT ?
                """, (f"%{query[:50]}%", limit)).fetchall()
                return [dict(r) for r in rows]

    def log_system_event(self, event_type: str, data: Dict) -> None:
        with self._lock:
            with self._conn() as conn:
                conn.execute("""
                    INSERT INTO system_events (event_type, data, created_at)
                    VALUES (?,?,?)
                """, (event_type, json.dumps(data, default=str)[:2000],
                      datetime.now().isoformat()))

    def get_memory_stats(self) -> Dict:
        with self._lock:
            with self._conn() as conn:
                total_memories = conn.execute(
                    "SELECT COUNT(*) FROM memories"
                ).fetchone()[0]
                total_tasks = conn.execute(
                    "SELECT COUNT(*) FROM tasks"
                ).fetchone()[0]
                total_errors = conn.execute(
                    "SELECT COUNT(*) FROM errors"
                ).fetchone()[0]
                return {
                    "total_memories": total_memories,
                    "total_tasks": total_tasks,
                    "total_errors": total_errors,
                    "session_id": self._session_id,
                }

    def end_session(self) -> None:
        with self._lock:
            with self._conn() as conn:
                conn.execute("""
                    UPDATE sessions SET ended_at=? WHERE session_id=?
                """, (datetime.now().isoformat(), self._session_id))

    def close(self) -> None:
        logger.info("MemorySystem closed.")
```

    event = {"type": event_type, "data": data, "timestamp": ...}
    # ... existing log/persist logic unchanged ...

        try:
            if inspect.iscoroutinefunction(handler):
                # Schedule on whatever loop is available
                try:
                    loop = asyncio.get_event_loop()
                    if loop.is_running():
                        asyncio.ensure_future(handler(event), loop=loop)
                    else:
                        loop.run_until_complete(handler(event))

"""
core/step_executor.py

Unified verify-retry step execution loop.
Every action across every agent goes through here.
See → Execute → Verify → Retry (with different strategy) → Escalate.
"""
import logging
import time
from dataclasses import dataclass, field
from typing import Dict, List, Optional, Tuple

from core.os_executor import ActionVerifier, FocusLostError, capture_region
from vision.screen_perception import ScreenPerception  # if exists, else use vision_system

logger = logging.getLogger("StepExecutor")


@dataclass
class StepResult:
    success: bool
    attempts: int = 0
    error: str = ""
    verified: bool = False
    recovery_strategies_used: List[str] = field(default_factory=list)
    elapsed: float = 0.0


class StepExecutor:
    """
    Wraps any callable action with:
    1. Pre-action screenshot
    2. Action execution
    3. Post-action screenshot comparison
    4. If no visual change AND change was expected → recovery → retry
    5. If max retries → return failure with diagnostic
    """

    def __init__(self, max_retries: int = 3, llm_router=None):
        self._max_retries = max_retries
        self._llm = llm_router

    def execute_with_verify(
        self,
        action_fn,           # callable — the actual pyautogui/UIA action
        canvas_region: Tuple,
        description: str,
        expect_visual_change: bool = True,
        recovery_fns: List = None,
    ) -> StepResult:
        """
        Execute action_fn, verify canvas changed, retry with recovery_fns on failure.
        recovery_fns: list of alternative callables tried in order on each failure.
        """
        start = time.time()
        result = StepResult(success=False)
        fns = [action_fn] + (recovery_fns or [])

        for attempt in range(self._max_retries):
            result.attempts = attempt + 1
            fn = fns[min(attempt, len(fns) - 1)]

            with ActionVerifier(region=canvas_region) as verifier:
                try:
                    fn()
                    time.sleep(0.2)  # UI settle

                    if expect_visual_change:
                        changed = verifier.verify_changed(description)
                        if not changed:
                            result.error = f"No visual change after: {description}"
                            logger.warning(result.error)
                            result.recovery_strategies_used.append(
                                f"attempt_{attempt + 1}_no_change"
                            )
                            time.sleep(0.3 * (attempt + 1))
                            continue

                    result.success = True
                    result.verified = True
                    result.elapsed = time.time() - start
                    return result

                except FocusLostError as e:
                    result.error = str(e)
                    logger.error(f"Focus lost on attempt {attempt + 1}: {e}")
                    time.sleep(0.5)

                except Exception as e:
                    result.error = str(e)
                    import pyautogui
                    try:
                        pyautogui.mouseUp()  # safety
                    except Exception:
                        pass
                    logger.error(f"Action failed attempt {attempt + 1}: {e}")
                    time.sleep(0.3)

        result.elapsed = time.time() - start
        return result
```

def _select_model(self, provider: Provider, task_type: str) -> str:
    models = provider.models
    if not models:
        return ""
    # Use different models for different task types when provider has multiple
    # Most providers list fastest→most capable
    MODEL_INDEX: Dict[str, int] = {
        "quick":   0,   # fastest model (index 0)
